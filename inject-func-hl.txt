 function hl(){ local cmd;local i;local color;i=1;cmd="cat";while [[ $# -gt 0 ]];do if [[ "${1}" != "." ]];then if [[ $i -lt 8 ]];then color="7\;$((30 + ${i}))";elif [[ $i -lt 14 ]];then color="$((30 - 7 + ${i}))";elif [[ $i -lt 21 ]];then color="7\;2\;$((30 - 13 + ${i}))";elif [[ $i -lt 28 ]];then color="4\;$((30 - 20 + ${i}))";fi;cmd="$cmd | GREP_COLOR="${color}" egrep --color=always --line-buffered '${1}|$'";fi;shift;i=$(($i+1));done;$SHELL -c "$cmd";}
